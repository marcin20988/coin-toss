4  /**
5  * \brief Class handling match scheduling
6  *
7  * Class handles division of players into classes, match scheduling and
8  * simulates coin toss games between scheduled pairs of players.
9  *
10  * Assignment of players into teams is completely random.
11  *
12  * Matchups are created in a way allowing maximum number of matches between
13  * players at the same time aiming for maximum level of randomness to the
14  * process and allowing for matchups between each individual pair of
15  * players.
16  *
17  * To achieve this the match assignment process is divided in two steps:
18  *
19  * 1) Most populated team is selected first and then any other team with
20  * more than one available player is selected and a matchup is created
21  * between two randomly selected players from these teams. This approach
22  * allows every pair of players to create a matchup at the same time
23  * ensuring that at the end of this stage each class retains a single player
24  * allowed for selection for the next step of the process.
25  *
26  * 2) This stage starts with all teams having one player available for
27  * selection therefore the pairs are created entirely randomly between
28  * ramaining players.
29  */

191  /**
192  * \param teams Array of teams to select from
193  *
194  * \param t Team already selected for the matchup
195  *
196  * \return Instance of team. Random team that can create a matchup with
197  * the first selected team $t
198  *
199  * In the match making process first the most populated team is chosen and
200  * the second team must be able to create a matchup with the first team.
201  *
202  * This function removes the first selected team from the list of
203  * available teams and also removes any other team that cannot create a
204  * matchup with team $t.
205  *
206  * For example if the only available pair of players between teams was
207  * chosen in the previous round there is no possible pair of players to
208  * form a matchup and the other team is removed from the list of available
209  * teams.
210  */
211  public function select_second_team($teams, $t)
212  {
213  // teams left for selection
214  $teams_left = $teams;
215 
216  // remove the team $t from the list. This is the team that has been
217  // already selected and matchup between players from the same teams are
218  // not allowed
219  $key = array_search($t, $teams_left);
220  unset($teams_left[$key]);
221  // update indices in the array
222  $teams_left = array_values($teams_left);
223 
224  // this is a list of teams that cannot create and allowed pair of
225  // players with team $t
226  $unpaired_teams = [];
227  foreach($teams_left as $t2)
228  {
229  // if teams cannot for any allowed matchup are scheduled to be deleted
230  if(!$t -> pairs_with($t2))
231  {
232  $unpaired_teams[] = array_search($t2, $teams_left);
233  }
234  }
235  // delete all "unpaired" teams and restructure the array
236  foreach($unpaired_teams as $i) unset($teams_left[$i]);
237  $teams_left = array_values($teams_left);
238 
239  $n = sizeof($teams_left);
240 
241  if($n > 0)
242  {
243  $i = rand(0, $n - 1);
244  return $teams_left[$i];
245  }

248  {
249  return False;

253  /**
254  * \return void
255  *
256  * \brief Schedule a single round of matched
257  *
258  * First all teams are reset for the round with all round-specific values
259  * set to defaults. Than the scheduling algorithm is invoked in two stages
260  * described in details in the class description.
261  */
262  public function schedule_next_round()
263  {
264  // matchups created in this round are stored here
265  $pairs = [];
266 
267  // reset all round-specific values
268  foreach($this -> teams as $t) $t -> new_round();
269 
270  // the algorithm has two steps and $i is the threshold value used for
271  // teams selection. In the first step we select players from teams that
272  // have more than one player (threhold $i = 1) and in the second step we
273  // take all teams with positive number of players (threshold $i=0)
274  for($i = 1; $i >=0 ; $i--)
275  {
276  // the loop breaks when there are no possible matchups left
277  while(true)
278  {
279  // teams having enough players to select from
280  $teams = $this -> get_available_teams($i);
281 
282  // first team is the one that has most unselected players left
283  $first_team = $this -> select_most_populated_team($teams);
284  if($first_team === False) break;
285 
286  // second team is selected randomly (with some restrictions - please
287  // look at the method comments)
288  $second_team = $this -> select_second_team($teams, $first_team);
289  if($second_team === False) break;
290 
291  // choose a pair of players from selected teams
292  $temp_pair = $this -> match_pair($first_team, $second_team);
293  $p1 = $temp_pair[0];
294  $p2 = $temp_pair[1];
295 
296  // save the current opponent for next round
297  $p1 -> previous_opponent = $p2 -> name;
298  $p2 -> previous_opponent = $p1 -> name;
299 
300  // assign heads or tails for each player randomly
301  $toss = rand(0, 1);
302  $p1 -> heads_tails[] = $toss ? "heads" : "tails";
303  $p2 -> heads_tails[] = !$toss ? "heads" : "tails";
304 
305  $pairs[] = array($p1, $p2);
306  }
307  }
308 
309  // check are there any players left unasigned to any match
310  $teams = $this -> get_available_teams(0);
311  $unpaired_players = [];
312  foreach($teams as $t)
313  {
314  foreach($t -> get_available_players() as $p)
315  {
316  // set some values to NULL so the length of heads_tails array is
317  // correct and clear the value of previous opponent
318  $p -> previous_opponent = NULL;
319  $p -> heads_tails[] = NULL;
320  $unpaired_players[] = $p;
321  }
322  }
323 
324  $this -> rounds[] = array("pairs" => $pairs, "unpaired_players" => $unpaired_players);
325  }
